FizzBuzz

Please enter two numbers between 1 and 100 and click "Solve FizzBuzz." The function will then cycle through all numbers between 1 and 100 and replace any numbers divisible by your first entered number with "Fizz", any numbers divisible by the second with "Buzz", and any divisible by both with "FizzBuzz."



//This is the beginning of the FizzBuzz function.
function fizzBuzz() {

        //This takes the numbers the user entered and stores them in variables.
        var fizz = Number($("#fbnum1").val());
        var buzz = Number($("#fbnum2").val());

        //This generates an array to display numbers 1-100.
        var fbArray = "";

        //The following two if statements ensure neither of the two numbers input
        //by the user are above 100 or below 0.
        if (fizz > 100 || buzz > 100) {
            alert("Number entered too high.")
            return
        }

        if (fizz < 1 || buzz < 1) {
            alert("Number entered too low.")
            return
        }

        //Loops through numbers 1-100 looking for which numbers within are
        //divisible by the user input numbers, then declares them as "Fizz",
        //"Buzz", or "FizzBuzz" accordingly if so.
        //Afterwards, stores all of this data in the previously set array.
        for (loop = 1; loop < 101; loop++) {
            if ((loop % fizz == 0) && (loop % buzz == 0)) {
                if (loop == 100) {
                    fbArray += "FizzBuzz";
                }
                else {
                    fbArray += "FizzBuzz" + ", ";
                }
            }

            else if (loop % fizz == 0) {
                if (loop == 100) {
                    fbArray += "Fizz";
                }
                else {
                    fbArray += "Fizz" + ", ";
                }
            }

            else if (loop % buzz == 0) {
                if (loop == 100) {
                    fbArray += "Buzz";
                }
                else {
                    fbArray += "Buzz" + ", ";
                }
            }

            else {
                if (loop == 0) {
                    fbArray += loop;
                }
                fbArray += loop + ", ";
            }

        }
        //Displays the results of the above code onscreen.
        $("#fboutput").show()
        $("#fboutput").html(`${fbArray}`);
    });
}

TacoCat

Type some text and click "Submit." This function will reverse your entered text and determine whether or not it is a palindrome.
i.e.: "tacocat" reversed is "tacocat."


//This is the beginning of the TacoCat function.
function tacoCat() {

        //This collects the user input text, and stores it in a variable.
        var str = ($("#tacoinput").val());

        //This calls an error if no text is entered by the user.
        if (str == "") {
            alert("No data entered.")
            return
        }

        //This takes the user input text and makes all letters entered in lowercase.
        var enteredText = str.toLowerCase();

        //This removes any spaces entered, allowing the user to enter palindromic
        //phrases such as "UFO Tofu."
        enteredText = enteredText.replace(/ /g, "");

        //The following variables determine the length of the entered text,
        //and then creates a variable to store the mirrored version of the text.
        var length = enteredText.length;

        var last = length - 1;

        var enteredTextRev = "";

        //This loops through the entered text, and then generates a mirrored
        //version to place in the above variable.
        for (loop = last; loop >= 0; loop--) {
            enteredTextRev += enteredText[loop];
        }

        //This makes a variable to compare the entered text with its mirror,
        //then displays a message confirming or denying its palindromic status.
        var results;

        if (enteredText === enteredTextRev) {
            results = `${str} is a palindrome!`
        }

        else {
            results = `${str} is not a palindrome!`
        }



        //This displays one of the above messages, depending on the
        //results of the function.
        $("#tacooutput").show();
        $("#tacooutput").html(`${results}`);

Sunset Hills

Enter heights for 6 buildings and click "Solve Sunset Hills." This function will then determine which buildings can see the sun setting to the left of them, and which buildings are obstructed by a building of equal or greater height.







//This is the beginning of the Sunset Hills function.
function sunsetHills() {

        //This generates an array from the user input building heights.
        let sunsetArray = new Array();
        sunsetArray.push(Number($("#sunsetnum1").val()));
        sunsetArray.push(Number($("#sunsetnum2").val()));
        sunsetArray.push(Number($("#sunsetnum3").val()));
        sunsetArray.push(Number($("#sunsetnum4").val()));
        sunsetArray.push(Number($("#sunsetnum5").val()));
        sunsetArray.push(Number($("#sunsetnum6").val()));

        //This ensures the first building will always see the sunset regardless of height,
        //as no buildings are between it and the sunset.
        let tallest = sunsetArray[0];

        //This forms a new array of the other buildings to later determine their comparative
        //heights and positions.
        let sunsetArray2 = new Array();
        sunsetArray2.push(tallest);

        //This loops through all entered building heights, skipping the first, as
        //no buildings are between it and the sunset.
        for (var loop = 1; loop < sunsetArray.length; loop++) {

            //In the event a building is taller than the buildings in front of it,
            //this pushes it into the array of buildings that can see the sunset.
            if (sunsetArray[loop] > tallest) {
                tallest = sunsetArray[loop];
                sunsetArray2.push(tallest);
            }
        };

        //This displays the results, listing all buildings that can see the sunset.
        $("#sunsetoutput").show()
        $("#sunsetoutput").html(`The buildings with heights of: ${sunsetArray2} can see the sunset.`)
}

Sum of All Fears

Enter a sum number, then enter up to 5 numbers and click "Submit" to determine if any combination of two of those numbers will add up to your sum.







//This is the beginning of the Sum of All Fears function.
function sumOfAllFears() {

        //This collects the top user input number and stores it in a variable.
        let topnum = Number($("#soaftopnum").val());

        //This generates an array and then collects the below user input numbers
        //and stores them into the array.
        let soafArray = [];
        soafArray.push(Number($("#soafnum1").val()));
        soafArray.push(Number($("#soafnum2").val()));
        soafArray.push(Number($("#soafnum3").val()));
        soafArray.push(Number($("#soafnum4").val()));
        soafArray.push(Number($("#soafnum5").val()));

        //This defaults the top number to 0 in the event no data was entered.
        //It also defaults the results as false until the loop is ran.
        let sumfears = 0;
        let found = false;

        //This loops through all potential sum combinations of the below numbers
        //and determines if any of them add up to the top number.
        for (var loop = 0; loop < soafArray.length; loop++) {

            sumfears = topnum - soafArray[loop];
            //In the event the top number is found as a sum of the below numbers,
            //this ends the loop function before cycling through the remaining loops.
            if (soafArray.indexOf(sumfears, loop + 1) > 0) {
                found = true;
                break;
            }


        }

        //This displays the results of the addition loop in the form of a message
        //stating whether or not any of the below numbers added up to the top number.
        $("#soafoutput").show()

        if (found)
            $("#soafoutput").html(`The sum of ${topnum} was found in the numbers entered.`);
        else
            $("#soafoutput").html(`The sum of ${topnum} was not found in the numbers entered.`);
}
